SELECT Long_Desc FROM FOOD_DES 
WHERE NDB_No IN (
	SELECT DISTINCT NDB_No FROM NUT_DATA
	WHERE Nutr_No = ( SELECT Nutr_No FROM NUTR_DEF
						WHERE NutrDesc = 'Water')
	EXCEPT SELECT DISTINCT NDB_No FROM NUT_DATA
	WHERE Nutr_No = ( SELECT Nutr_No FROM NUTR_DEF
						WHERE NutrDesc = 'Calcium, Ca')
)
ORDER BY FdGrp_Cd DESC, Shrt_Desc ASC
;
